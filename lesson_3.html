1. С помощью цикла while вывести все простые числа в промежутке от 0 до 100.


function simpleNum() {
var num = 2;
while (num <= 100) {
if (simple(num)) {
console.log(num + " ");
}
num++;
}
function simple(num) {
var i = 2
; while (i < num) {
if (num % i == 0) {
return false;
}
i++;
}
return true;
}
}
